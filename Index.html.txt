<script>
    const bell = document.getElementById("bell-sound");
    const music = document.getElementById("bg-music");

    // Autoplay Fix
    window.addEventListener("click", function(){
        bell.play();
        music.play();
    }, { once:true });

    const weddingDate=new Date("April 12, 2026 06:30:00").getTime();
    const countdownEl=document.getElementById("countdown");

    const interval=setInterval(function(){
        const now=new Date().getTime();
        const distance=weddingDate-now;

        if(distance>0){
            const days=Math.floor(distance/(1000*60*60*24));
            const hours=Math.floor((distance%(1000*60*60*24))/(1000*60*60));
            const minutes=Math.floor((distance%(1000*60*60))/(1000*60));
            const seconds=Math.floor((distance%(1000*60))/1000);
            countdownEl.innerHTML=days+"d "+hours+"h "+minutes+"m "+seconds+"s to go";
        } else {
            countdownEl.innerHTML="The Wedding Day is Here!";
            clearInterval(interval);
            startFireworks();
        }
    },1000);

    // Simple Fireworks
    const canvas=document.getElementById("fireworks");
    const ctx=canvas.getContext("2d");
    canvas.width=window.innerWidth;
    canvas.height=window.innerHeight;

    function startFireworks(){
        for(let i=0;i<100;i++){
            setTimeout(()=>{
                const x=Math.random()*canvas.width;
                const y=Math.random()*canvas.height/2;
                for(let j=0;j<50;j++){
                    ctx.beginPath();
                    ctx.arc(x+Math.random()*100-50,y+Math.random()*100-50,2,0,Math.PI*2);
                    ctx.fillStyle="gold";
                    ctx.fill();
                }
            },i*100);
        }
    }
</script>
